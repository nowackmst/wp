Save wallet and validator address

WALLET_ADDRESS=$(wardend keys show $WALLET -a) VALOPER_ADDRESS=$(wardend keys show $WALLET --bech val -a) echo "export WALLET_ADDRESS="$WALLET_ADDRESS >> $HOME/.bash_profile echo "export VALOPER_ADDRESS="$VALOPER_ADDRESS >> $HOME/.bash_profile source $HOME/.bash_profile
